<script>
    // ---------- STATE MANAGEMENT ----------
    let currentDate = new Date();
    let selectedDay = null;      // stores day number (1-30)
    let selectedTimeSlot = null; // stores string like "09:00 AM"
    let selectedReason = null;
    let selectedNurse = null;

    // Preset time slots (clinic vibe)
    const defaultTimeSlots = ["08:30 AM", "09:00 AM", "10:00 AM", "11:00 AM", "01:00 PM", "02:30 PM", "03:30 PM", "04:00 PM"];
    // Reason options
    const reasonsList = ["Check-up", "Medical Clearance", "Follow-up", "Lab Results Reading", "Vaccination", "Others"];
    // Personnel / Nurses
    const nursesList = ["Nurse Marilou G.", "Nurse Jonathan C.", "Nina Reyes (RN)", "Christine Dela Cruz", "Dr. Alvin (Duty MD)"];

    // DOM elements
    const monthYearDisplay = document.getElementById("monthYearDisplay");
    const daysGrid = document.getElementById("daysGrid");
    const timeSlotsContainer = document.getElementById("timeSlotsContainer");
    const reasonsGrid = document.getElementById("reasonsGrid");
    const personnelGrid = document.getElementById("personnelGrid");
    const confirmBtn = document.getElementById("confirmAppointmentBtn");
    const goBackBtn = document.getElementById("goBackBtn");
    const faqBtn = document.getElementById("faqBtn");
    const customTimeInput = document.getElementById("customTimeInput");
    const applyCustomTimeBtn = document.getElementById("applyCustomTimeBtn");
    const liveTimeDisplay = document.getElementById("liveTimeDisplay");

    // Helper: show temporary message
    function showToast(msg, duration = 2200) {
        const toast = document.getElementById("toastMessage");
        toast.textContent = msg;
        toast.style.visibility = "visible";
        toast.style.opacity = "1";
        setTimeout(() => {
            toast.style.opacity = "0";
            setTimeout(() => { toast.style.visibility = "hidden"; }, 300);
        }, duration);
    }

    // Generate days 1-30 (showing current month/year)
    function renderDatePicker() {
        const months = ["January", "February", "March", "April", "May", "June", "July", "August", "September", "October", "November", "December"];
        const year = currentDate.getFullYear();
        const month = currentDate.getMonth();
        monthYearDisplay.textContent = `${months[month]} ${year}`;

        daysGrid.innerHTML = "";
        // Days 1 to 30 (kiosk requirement: up to 30)
        for (let i = 1; i <= 30; i++) {
            const dayBtn = document.createElement("button");
            dayBtn.textContent = i;
            dayBtn.classList.add("day-btn");
            if (selectedDay === i) dayBtn.classList.add("selected");
            dayBtn.addEventListener("click", () => {
                selectedDay = i;
                renderDatePicker();
                // when date changes, keep time selection? optional reset slot selection? better not reset time completely but keep UX
                // but we keep selectedTimeSlot, but we do not auto-clear.
                showToast(`Date set to ${months[month]} ${i}, ${year}`, 1200);
            });
            daysGrid.appendChild(dayBtn);
        }
    }

    // render time slots (stat card buttons)
    function renderTimeSlots() {
        timeSlotsContainer.innerHTML = "";
        defaultTimeSlots.forEach(slot => {
            const btn = document.createElement("button");
            btn.textContent = slot;
            btn.classList.add("time-slot-btn");
            if (selectedTimeSlot === slot) btn.classList.add("selected-time");
            btn.addEventListener("click", () => {
                selectedTimeSlot = slot;
                renderTimeSlots();   // re-render to update selection
                showToast(`Time slot: ${slot} selected`, 1000);
            });
            timeSlotsContainer.appendChild(btn);
        });
    }

    // reason buttons
    function renderReasons() {
        reasonsGrid.innerHTML = "";
        reasonsList.forEach(reason => {
            const btn = document.createElement("button");
            btn.textContent = reason;
            btn.classList.add("reason-btn");
            if (selectedReason === reason) btn.classList.add("selected-reason");
            btn.addEventListener("click", () => {
                selectedReason = reason;
                renderReasons();
                showToast(`Reason: ${reason}`, 1000);
            });
            reasonsGrid.appendChild(btn);
        });
    }

    // nurse personnel
    function renderPersonnel() {
        personnelGrid.innerHTML = "";
        nursesList.forEach(nurse => {
            const nurseBtn = document.createElement("button");
            nurseBtn.textContent = nurse;
            nurseBtn.classList.add("nurse-card-btn");
            if (selectedNurse === nurse) nurseBtn.classList.add("selected-nurse");
            nurseBtn.addEventListener("click", () => {
                selectedNurse = nurse;
                renderPersonnel();
                showToast(`Assigned: ${nurse}`, 1000);
            });
            personnelGrid.appendChild(nurseBtn);
        });
    }

    // confirm appointment check and summary
    function confirmAppointment() {
        if (!selectedDay) {
            showToast("❌ Please select a date (1-30).", 1800);
            return;
        }
        if (!selectedTimeSlot && !customTimeInput.value.trim()) {
            showToast("⏰ Please select a time slot or enter a custom time.", 1800);
            return;
        }
        if (!selectedReason) {
            showToast("📋 Please select a reason for visit.", 1800);
            return;
        }
        if (!selectedNurse) {
            showToast("👩‍⚕️ Please select an assigned nurse / personnel.", 1800);
            return;
        }
        const finalTime = selectedTimeSlot ? selectedTimeSlot : customTimeInput.value.trim();
        if (!finalTime) {
            showToast("Please provide time slot.", 1800);
            return;
        }
        const monthNames = ["January", "February", "March", "April", "May", "June", "July", "August", "September", "October", "November", "December"];
        const year = currentDate.getFullYear();
        const month = currentDate.getMonth();
        const dateString = `${monthNames[month]} ${selectedDay}, ${year}`;
        const msg = `✅ Appointment confirmed!\n📅 Date: ${dateString}\n⏰ Time: ${finalTime}\n📌 Reason: ${selectedReason}\n🧑‍⚕️ Nurse: ${selectedNurse}`;
        alert(msg);  // classic kiosk confirm
        showToast("Appointment recorded! Thank you.", 2000);
        // optional: could reset form, but preserve till next action? we might keep but fine.
        // Prefer not auto-reset to allow multiple? but better reset? We'll keep so they can change.
    }

    // custom time button handler
    function setCustomTime() {
        const customVal = customTimeInput.value.trim();
        if (customVal === "") {
            showToast("Please enter a time (e.g., 2:30 PM)", 1200);
            return;
        }
        selectedTimeSlot = null; // deselect predefined
        renderTimeSlots();  // unhighlight any preset slot
        showToast(`Custom time: ${customVal} set`, 1200);
        // store custom time in separate variable for confirmation
        // We'll store custom selected flag, but confirm uses customTimeInput value.
    }

    // go back simulation (just toast & a little history behaviour, but in kiosk we show feedback)
    function goBackAction() {
        showToast("🔙 Returning to kiosk main menu...", 1500);
        // preserve same design but can mimic back action: in real kiosk could redirect to dashboard.
        // For now just message, but doesn't violate design, plus keep UI.
        setTimeout(() => {
            // optional gentle reset? we'll just notify that back navigation would reload but not required
        }, 400);
    }

    function faqAction() {
        alert("📞 Need more help?\n\n• For assistance, please approach the clinic front desk.\n• You may call PUP Clinic Hotline: (02) 8335-1786\n• Or email: clinic@pup.edu.ph\n\nFAQ: Walk-in available Mon-Fri 8AM-5PM.");
        showToast("FAQ – additional help displayed", 1500);
    }

    // Live clock update (still have the time in the upper part)
    function updateLiveClock() {
        const now = new Date();
        let hours = now.getHours();
        const minutes = now.getMinutes().toString().padStart(2,'0');
        const ampm = hours >= 12 ? 'PM' : 'AM';
        hours = hours % 12 || 12;
        liveTimeDisplay.textContent = `${hours}:${minutes} ${ampm}`;
    }
    setInterval(updateLiveClock, 1000);
    updateLiveClock();

    // combine all renders and attach listeners
    function initUI() {
        renderDatePicker();
        renderTimeSlots();
        renderReasons();
        renderPersonnel();
        confirmBtn.addEventListener("click", confirmAppointment);
        goBackBtn.addEventListener("click", goBackAction);
        faqBtn.addEventListener("click", faqAction);
        applyCustomTimeBtn.addEventListener("click", setCustomTime);
    }

    initUI();

    // ensure that middle section uses max space, no forced scroll due to internal limited content
    // we made days grid & time slots scroll only inside their containers if many, but overall container won't have body scroll.
    // The design respects "maximize space between separators"
    window.addEventListener('resize', () => {});
</script>
